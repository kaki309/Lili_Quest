#include <SPI.h>
#include <MFRC522.h>
#include <ArduinoJson.h>

// ==== RFID ====
#define SS_PIN 10
#define RST_PIN 9
MFRC522 mfrc522(SS_PIN, RST_PIN);

// ==== ENTRADAS ====
int pinX = A0;
int pinY = A1;
int pinPOT = A2;
int pinBTN = A3;
int pinSW = 2;

// ==== CONTROL ====
bool conectadoUnity = false;
bool tarjetaDetectada = false;

unsigned long lastAnnounceTime = 0;

// ==== VARIABLES CAMBIO ====
int lastX = 0;
int lastY = 0;
int lastPOT = 0;
int lastBTN = HIGH;

// ==== RFID PERSISTENTE ====
String lastRFID = "";

// ==== LEER UID ====
String leerUID() {
  String uid = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    uid += String(mfrc522.uid.uidByte[i], HEX);
  }
  uid.toUpperCase();
  return uid;
}

void setup() {
  Serial.begin(9600);

  SPI.begin();
  mfrc522.PCD_Init();

  pinMode(pinBTN, INPUT_PULLUP);
  pinMode(pinSW, INPUT_PULLUP);

  Serial.println("esperando conexion con unity...");
}

void loop() {

  // =========================================
  // 1. ESPERAR UNITY
  // =========================================
  if (!conectadoUnity) {

    unsigned long now = millis();
    if (now - lastAnnounceTime > 1000) {
      Serial.println("soy controles lili quest");
      lastAnnounceTime = now;
    }

    if (Serial.available() > 0) {
      String msg = Serial.readStringUntil('\n');
      msg.trim();

      Serial.print("recibido: ");
      Serial.println(msg);

      msg.toLowerCase();

      if (msg.indexOf("te encontre") >= 0) {
        conectadoUnity = true;
        Serial.println("conectado con unity");
        Serial.println("acerca una tarjeta rfid...");
      }
    }

    return;
  }

  // =========================================
  // 2. ESPERAR RFID
  // =========================================
  if (!tarjetaDetectada) {

    if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {

      lastRFID = leerUID();

      Serial.print("tarjeta detectada: ");
      Serial.println(lastRFID);

      tarjetaDetectada = true;

      Serial.println("sistema activado");

      mfrc522.PICC_HaltA();
      mfrc522.PCD_StopCrypto1();
    }

    return;
  }

  // =========================================
  // 3. LECTURA DE SENSORES
  // =========================================
  int xValue = analogRead(pinX);
  int yValue = analogRead(pinY);
  int potValue = analogRead(pinPOT);
  int btnValue = digitalRead(pinBTN);

  // =========================================
  // 4. DEBUG EN MONITOR
  // =========================================
  if (abs(xValue - lastX) > 20 || abs(yValue - lastY) > 20) {
    Serial.print("joystick -> x:");
    Serial.print(xValue);
    Serial.print(" y:");
    Serial.println(yValue);

    lastX = xValue;
    lastY = yValue;
  }

  if (abs(potValue - lastPOT) > 20) {
    Serial.print("potenciometro: ");
    Serial.println(potValue);
    lastPOT = potValue;
  }

  if (btnValue != lastBTN) {
    if (btnValue == LOW) {
      Serial.println("boton presionado");
    } else {
      Serial.println("boton suelto");
    }
    lastBTN = btnValue;
  }

  // =========================================
  // 5. ENVÍO A UNITY (JSON)
  // =========================================
  StaticJsonDocument<200> doc;

  doc["RFID"] = lastRFID;
  doc["JOYSTICK"] = String(xValue) + "-" + String(yValue);
  doc["POT"] = String(potValue);
  doc["BUTTON"] = (btnValue == LOW ? "P" : "S");

  String output;
  serializeJson(doc, output);

  Serial.println(output);

  delay(100);

  // =========================================
  // 6. COMANDOS DESDE UNITY
  // =========================================
  if (Serial.available() > 0) {
    String msg = Serial.readStringUntil('\n');
    msg.trim();

    msg.toLowerCase();

    if (msg.indexOf("reset") >= 0) {
      Serial.println("reiniciando sistema...");
      conectadoUnity = false;
      tarjetaDetectada = false;
      lastRFID = "";
    }
  }
}
